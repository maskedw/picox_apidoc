449 xilist_swap(XIntrusiveList* self, XIntrusiveList* other)
450 {
451  X_ASSERT(self);
452  X_ASSERT(other);
453 
454  if (self == other)
455  return;
456 
457  const bool empty_self = xilist_empty(self);
458  const bool empty_other = xilist_empty(other);
459 
460  X_SWAP(self->head, other->head, XIntrusiveNode);
461 
462  if (empty_other)
463  xilist_init(self);
464  else
465  self->head.next->prev = self->head.prev->next = &(self->head);
466 
467  if (empty_self)
468  xilist_init(other);
469  else
470  other->head.next->prev = other->head.prev->next = &(other->head);
471 }
