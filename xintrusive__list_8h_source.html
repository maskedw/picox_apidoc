475 X_INLINE void
476 xilist_swap(XIntrusiveList* self, XIntrusiveList* other)
477 {
478  bool empty_self, empty_other;
479  X_ASSERT(self);
480  X_ASSERT(other);
481 
482  if (self == other)
483  return;
484 
485  empty_self = xilist_empty(self);
486  empty_other = xilist_empty(other);
487 
488  X_SWAP(self->head, other->head, XIntrusiveNode);
489 
490  if (empty_other)
491  xilist_init(self);
492  else
493  self->head.next->prev = self->head.prev->next = &(self->head);
494 
495  if (empty_self)
496  xilist_init(other);
497  else
498  other->head.next->prev = other->head.prev->next = &(other->head);
499 }
